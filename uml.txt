enum Status
{
    Running
    Failure
    Success
}

interface INode
{
    + Status Process(name: ref string)
}

interface IResettable
{
    + void Reset();
}

interface IBlackboardRequired
{
    + Blackboard: Blackboard { get; set; }
    + void AssignBlackboard(blackboard: Blackboard);
}

class Sense<T> 
{

    + Status Process(name: ref string)

}
