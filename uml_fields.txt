@startuml

abstract class Field
{
    + key: string
    + virtual void Setup(blackboard: Blackboard)
    + abstract void GetFromBlackboard()
}

class StringField extends Field
{
    + override void Setup(blackboard: Blackboard)
    + override void GetFromBlackboard()
    - void SendToBlackboard(str: string)
}

class NumberField extends Field
{
    + override void Setup(blackboard: Blackboard)
    + override void GetFromBlackboard()
    - void SendToBlackboard(value: float)
}

@enduml
